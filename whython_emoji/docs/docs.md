# If statements
```
if value == value do
    # do something
elif value == other_value do
    # do something
else
    # do something
end
```

# While loop
```
while value == value do
    # do something
    break/return/continue
end
```

# For loop
``` 
for var i = 0 to 5 step 1 do
    print(i)
    break/return/continue
end
```
